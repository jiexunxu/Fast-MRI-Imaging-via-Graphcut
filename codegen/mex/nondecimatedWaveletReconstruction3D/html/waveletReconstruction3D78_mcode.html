 8function B=waveletReconstruction3D(A, Lo_R, Hi_R)
 9    [m, n, p]=size(A);p=p/8;
10    q=length(Lo_R);L=size(A, 3)/8;
11    B=zeros(m+q-1, n+q-1, p+q-1);    
12    B=B+singleBlockRecon(A(:, :, 1:L), Lo_R, Lo_R, Lo_R);
13    B=B+singleBlockRecon(A(:, :, L+1:2*L), Hi_R, Lo_R, Lo_R);
14    B=B+singleBlockRecon(A(:, :, 2*L+1:3*L), Lo_R, Hi_R, Lo_R);
15    B=B+singleBlockRecon(A(:, :, 3*L+1:4*L), Hi_R, Hi_R, Lo_R);
16    B=B+singleBlockRecon(A(:, :, 4*L+1:5*L), Lo_R, Lo_R, Hi_R);
17    B=B+singleBlockRecon(A(:, :, 5*L+1:6*L), Hi_R, Lo_R, Hi_R);
18    B=B+singleBlockRecon(A(:, :, 6*L+1:7*L), Lo_R, Hi_R, Hi_R);
19    B=B+singleBlockRecon(A(:, :, 7*L+1:8*L), Hi_R, Hi_R, Hi_R);    
20    B=B(q:m, q:n, q:p);
21end
