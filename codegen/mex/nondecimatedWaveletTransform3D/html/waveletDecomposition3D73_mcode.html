 8% LLL, LLH, LHL, LHH, HLL, HLH, HHL, HHH
 9function B=waveletDecomposition3D(A, Lo_D, Hi_D)
10    [m, n, p]=size(A);
11    q=length(Lo_D);
12    B=complex(zeros(m+q-1, n+q-1, (p+q-1)*8), 0);    
13    BL=directionalConv(A, Lo_D, 1);
14    BH=directionalConv(A, Hi_D, 1);
15    BLL=directionalConv(BL, Lo_D, 2);
16    BLH=directionalConv(BL, Hi_D, 2);
17    BHL=directionalConv(BH, Lo_D, 2);
18    BHH=directionalConv(BH, Hi_D, 2);
19    L=p+q-1;
20    B(:, :, 1:L)=directionalConv(BLL, Lo_D, 3);
21    B(:, :, L+1:2*L)=directionalConv(BLL, Hi_D, 3);
22    B(:, :, 2*L+1:3*L)=directionalConv(BLH, Lo_D, 3);
23    B(:, :, 3*L+1:4*L)=directionalConv(BLH, Hi_D, 3);
24    B(:, :, 4*L+1:5*L)=directionalConv(BHL, Lo_D, 3);
25    B(:, :, 5*L+1:6*L)=directionalConv(BHL, Hi_D, 3);
26    B(:, :, 6*L+1:7*L)=directionalConv(BHH, Lo_D, 3);
27    B(:, :, 7*L+1:8*L)=directionalConv(BHH, Hi_D, 3);
28end
