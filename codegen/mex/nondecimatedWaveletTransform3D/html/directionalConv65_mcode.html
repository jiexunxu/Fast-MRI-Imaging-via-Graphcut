30function B=directionalConv(A, w, dir)
31    [m, n, p]=size(A);
32    q=length(w);
33    if dir==1   
34        B=complex(zeros(m+q-1, n, p), 0);
35        for j=1:n
36            for k=1:p
37                v=conv(squeeze(A(:, j, k)), w);
38                B(:, j, k)=v;
39            end
40        end
41    elseif dir==2
42        B=complex(zeros(m, n+q-1, p), 0);
43        for i=1:m
44            for k=1:p
45                v=conv(squeeze(A(i, :, k)), w);
46                B(i, :, k)=v;
47            end
48        end
49    elseif dir==3 
50        B=complex(zeros(m, n, p+q-1), 0);
51        for i=1:m
52            for j=1:n
53                v=conv(squeeze(A(i, j, :)), w);
54                B(i, j, :)=v;
55            end
56        end
57    end
58end
