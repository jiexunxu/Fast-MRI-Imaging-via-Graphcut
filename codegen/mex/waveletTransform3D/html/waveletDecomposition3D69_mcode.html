 12% LLL, LLH, LHL, LHH, HLL, HLH, HHL, HHH
 13function B=waveletDecomposition3D(A, Lo_D, Hi_D)
 14    [m, n, p]=size(A);
 15    q=length(Lo_D);
 16    B=complex(zeros(m+q-1, n+q-1, (p+q-1)*8), 0);    
 17    BL=directionalConv(A, Lo_D, 1);
 18    BH=directionalConv(A, Hi_D, 1);
 19    BLL=directionalConv(BL, Lo_D, 2);
 20    BLH=directionalConv(BL, Hi_D, 2);
 21    BHL=directionalConv(BH, Lo_D, 2);
 22    BHH=directionalConv(BH, Hi_D, 2);
 23    L=p+q-1;
 24    B(:, :, 1:L)=directionalConv(BLL, Lo_D, 3);
 25    B(:, :, L+1:2*L)=directionalConv(BLL, Hi_D, 3);
 26    B(:, :, 2*L+1:3*L)=directionalConv(BLH, Lo_D, 3);
 27    B(:, :, 3*L+1:4*L)=directionalConv(BLH, Hi_D, 3);
 28    B(:, :, 4*L+1:5*L)=directionalConv(BHL, Lo_D, 3);
 29    B(:, :, 5*L+1:6*L)=directionalConv(BHL, Hi_D, 3);
 30    B(:, :, 6*L+1:7*L)=directionalConv(BHH, Lo_D, 3);
 31    B(:, :, 7*L+1:8*L)=directionalConv(BHH, Hi_D, 3);
 32end
