 50function B=directionalConv(A, w, dir)
 51    [m, n, p]=size(A);
 52    q=length(w);
 53    if dir==1   
 54        B=complex(zeros(m+q-1, n, p), 0);
 55        for j=1:n
 56            for k=1:p
 57                v=conv(squeeze(A(:, j, k)), w);
 58                B(:, j, k)=v;
 59            end
 60        end
 61    elseif dir==2
 62        B=complex(zeros(m, n+q-1, p), 0);
 63        for i=1:m
 64            for k=1:p
 65                v=conv(squeeze(A(i, :, k)), w);
 66                B(i, :, k)=v;
 67            end
 68        end
 69    elseif dir==3 
 70        B=complex(zeros(m, n, p+q-1), 0);
 71        for i=1:m
 72            for j=1:n
 73                v=conv(squeeze(A(i, j, :)), w);
 74                B(i, j, :)=v;
 75            end
 76        end
 77    end
 78end
