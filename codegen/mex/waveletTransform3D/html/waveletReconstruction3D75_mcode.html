 34function B=waveletReconstruction3D(A, Lo_R, Hi_R)
 35    [m, n, p]=size(A);p=p/8;
 36    q=length(Lo_R);
 37    L=p+q-1;
 38    B=zeros(m+q-1, n+q-1, p+q-1);    
 39    B=B+singleBlockRecon(A(:, :, 1:L), Lo_R, Lo_R, Lo_R);
 40    B=B+singleBlockRecon(A(:, :, L+1:2*L), Hi_R, Lo_R, Lo_R);
 41    B=B+singleBlockRecon(A(:, :, 2*L+1:3*L), Lo_R, Hi_R, Lo_R);
 42    B=B+singleBlockRecon(A(:, :, 3*L+1:4*L), Hi_R, Hi_R, Lo_R);
 43    B=B+singleBlockRecon(A(:, :, 4*L+1:5*L), Lo_R, Lo_R, Hi_R);
 44    B=B+singleBlockRecon(A(:, :, 5*L+1:6*L), Hi_R, Lo_R, Hi_R);
 45    B=B+singleBlockRecon(A(:, :, 6*L+1:7*L), Lo_R, Hi_R, Hi_R);
 46    B=B+singleBlockRecon(A(:, :, 7*L+1:8*L), Hi_R, Hi_R, Hi_R);    
 47    B=B(q:m, q:n, q:p);
 48end
